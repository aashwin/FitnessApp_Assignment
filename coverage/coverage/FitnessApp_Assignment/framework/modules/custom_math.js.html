    var longDiffRadians = (long2 - long1) * Math.PI / 180;
 
    var a = Math.sin(latDiffRadians / 2) * Math.sin(latDiffRadians / 2) +
        Math.cos(lat1Radians) * Math.cos(lat2Radians) *
